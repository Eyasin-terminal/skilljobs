        document.addEventListener('DOMContentLoaded', () => {
            
            // Countdown Logic for the Form Header
            let time = 2 * 24 * 60 * 60 + 14 * 3600 + 45 * 60 + 30; // 2 Days, 14 Hours, 45 Mins, 30 Secs
            
            const daysEl = document.getElementById('days');
            const hoursEl = document.getElementById('hours');
            const minsEl = document.getElementById('mins');
            const secsEl = document.getElementById('secs');
            
            if(daysEl && hoursEl && minsEl && secsEl) {
                setInterval(() => {
                    if(time > 0) time--;
                    
                    let d = Math.floor(time / (24 * 3600));
                    let h = Math.floor((time % (24 * 3600)) / 3600);
                    let m = Math.floor((time % 3600) / 60);
                    let s = Math.floor(time % 60);
                    
                    daysEl.innerText = d.toString().padStart(2, '0');
                    hoursEl.innerText = h.toString().padStart(2, '0');
                    minsEl.innerText = m.toString().padStart(2, '0');
                    secsEl.innerText = s.toString().padStart(2, '0');
                }, 1000);
            }

            // Form Submit Simulation
            const form = document.getElementById('enrollmentForm');
            if(form) {
                form.addEventListener('submit', (e) => {
                    e.preventDefault();
                    const btn = form.querySelector('button');
                    const originalText = btn.innerHTML;
                    
                    btn.innerHTML = 'Sending Details...';
                    btn.classList.add('opacity-75', 'cursor-not-allowed');
                    
                    setTimeout(() => {
                        btn.innerHTML = 'Registration Successful! ✓';
                        // Success state styling
                        btn.classList.replace('from-[#00E5FF]', 'from-[#10B981]');
                        btn.classList.replace('to-[#0077FF]', 'to-[#059669]');
                        btn.classList.replace('hover:from-[#00C2FF]', 'hover:from-[#34D399]');
                        btn.classList.replace('hover:to-[#0055FF]', 'hover:to-[#047857]');
                        btn.classList.replace('shadow-[0_4px_20px_rgba(0,229,255,0.4)]', 'shadow-[0_4px_20px_rgba(16,185,129,0.4)]');
                        
                        setTimeout(() => {
                            form.reset();
                            btn.innerHTML = originalText;
                            btn.classList.remove('opacity-75', 'cursor-not-allowed');
                            btn.classList.replace('from-[#10B981]', 'from-[#00E5FF]');
                            btn.classList.replace('to-[#059669]', 'to-[#0077FF]');
                            btn.classList.replace('hover:from-[#34D399]', 'hover:from-[#00C2FF]');
                            btn.classList.replace('hover:to-[#047857]', 'hover:to-[#0055FF]');
                            btn.classList.replace('shadow-[0_4px_20px_rgba(16,185,129,0.4)]', 'shadow-[0_4px_20px_rgba(0,229,255,0.4)]');
                        }, 3000);
                    }, 1500);
                });
